\newcommand{\drawcube}[4]
{
	% Bottom
	\draw[fill=darker#4] (#1,#2,#3) -- (#1+1,#2,#3) -- (#1+1,#2,#3+1) -- (#1,#2,#3+1) -- cycle;
	
	% Front & back
	%\draw[] (#1,#2,#3) -- (#1+1,#2,#3) -- (#1+1,#2+1,#3) -- (#1,#2+1,#3) -- cycle;
	\draw[fill=dark#4] (#1,#2,#3) -- (#1+1,#2,#3) -- (#1+1,#2+1,#3) -- (#1,#2+1,#3) -- cycle;
	\draw[fill=dark#4] (#1,#2,#3+1) -- (#1+1,#2,#3+1) -- (#1+1,#2+1,#3+1) -- (#1,#2+1,#3+1) -- cycle;
	
	% Left & right
	%\draw[] (#1,#2,#3) -- (#1,#2,#3+1) -- (#1,#2+1,#3+1) -- (#1,#2+1,#3) -- cycle;
	\draw[fill=darker#4] (#1,#2,#3) -- (#1,#2,#3+1) -- (#1,#2+1,#3+1) -- (#1,#2+1,#3) -- cycle;
	\draw[fill=darker#4] (#1+1,#2,#3) -- (#1+1,#2,#3+1) -- (#1+1,#2+1,#3+1) -- (#1+1,#2+1,#3) -- cycle;
	
	% Top & bottom
	\draw[fill=darker#4] (#1,#2+1,#3) -- (#1+1,#2+1,#3) -- (#1+1,#2+1,#3+1) -- (#1,#2+1,#3+1) -- cycle;
}
